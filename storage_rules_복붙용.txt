rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    // 📷 프로필 사진 (2026-07-23 추가) — 본인 uid 파일명(uid.jpg)만 업로드, 이미지 5MB 제한
    match /profile-photos/{fileName} {
      allow read: if request.auth != null;
      allow write: if request.auth != null
                   && fileName == request.auth.uid + '.jpg'
                   && request.resource.contentType.matches('image/.*')
                   && request.resource.size < 5 * 1024 * 1024;
    }

    match /equipment/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    match /community/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    match /facilities/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    match /signatures/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    match /profiles/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    match /return_photos/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    match /attachments/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    match /external-rentals/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    match /scenepatch/{file}{
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
